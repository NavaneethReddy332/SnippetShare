    <script>
        const overlay = document.getElementById('overlay');
        const slider = document.getElementById('slider');
        const loginPanel = document.getElementById('loginPanel');
        const signupPanel = document.getElementById('signupPanel');

        function toggleModal(show) {
            if(show) {
                overlay.classList.add('active');
            } else {
                overlay.classList.remove('active');
            }
        }

        overlay.addEventListener('click', (e) => {
            if (e.target === overlay) toggleModal(false);
        });

        function slide(target) {
            if (target === 'signup') {
                // Move slider to the left
                slider.style.transform = 'translateX(-50%)';
                
                // Manage stagger classes
                loginPanel.classList.remove('active-panel');
                
                // Small delay to start signup animation so it looks like it's entering fresh
                setTimeout(() => {
                    signupPanel.classList.add('active-panel');
                }, 100);

            } else {
                // Move slider back to start
                slider.style.transform = 'translateX(0)';
                
                // Manage stagger classes
                signupPanel.classList.remove('active-panel');
                
                setTimeout(() => {
                    loginPanel.classList.add('active-panel');
                }, 100);
            }
        }
    </script>
